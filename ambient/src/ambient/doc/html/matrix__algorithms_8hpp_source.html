00044     bool is_hermitian(const matrix<T,A>& a)
00045     {
00046         if(num_rows(a) != num_cols(a))
00047             return false;
00048         for (size_t i=0; i < num_rows(a); ++i)
00049             for(size_t j=0; j < num_cols(a); ++j){
00050                if( a(i,j) != ambient::numeric::kernels::helper_complex<T>::conj(a(j,i)) )
00051                    return false;
00052             }
00053         return true;
00054     }
